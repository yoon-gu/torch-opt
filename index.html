 1import torch
 2import torch.nn as nn
 3import torch.nn.functional as F
 4
 5class QNetwork(nn.Module):
 6    def __init__(self, state_size, action_size, seed,
 7                 fc1_units=256, fc2_units=128, fc3_units=64):
 8        super(QNetwork, self).__init__()
 9        self.seed = torch.manual_seed(seed)
10        self.fc1 = nn.Linear(state_size, fc1_units)
11        self.fc2 = nn.Linear(fc1_units, fc2_units)
12        self.fc3 = nn.Linear(fc2_units, fc3_units)
13        self.fc4 = nn.Linear(fc3_units, action_size)
14
15    def forward(self, state):
16        x = F.relu(self.fc1(state))
17        x = F.relu(self.fc2(x))
18        x = F.relu(self.fc3(x))
19        return self.fc4(x)
